-- Instrucciones:
-- Ejecutar el siguiente script SQL en el SQL Editor de Supabase para crear la tabla de Planes de Mantenimiento

CREATE TABLE IF NOT EXISTS public.planes_mantenimiento (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  activo_tag text NOT NULL,
  tarea text NOT NULL,
  frecuencia_dias integer NOT NULL,
  tecnico_asignado text,
  ultima_ejecucion date,
  proxima_fecha date NOT NULL,
  checklist jsonb DEFAULT '[]'::jsonb,
  created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL,
  updated_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Políticas de RLS (Row Level Security)
-- Permitimos acceso total al API
ALTER TABLE public.planes_mantenimiento ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Enable read access for all users on planes_mantenimiento"
ON public.planes_mantenimiento FOR SELECT
USING (true);

CREATE POLICY "Enable insert access for all users on planes_mantenimiento"
ON public.planes_mantenimiento FOR INSERT
WITH CHECK (true);

CREATE POLICY "Enable update access for all users on planes_mantenimiento"
ON public.planes_mantenimiento FOR UPDATE
USING (true);

CREATE POLICY "Enable delete access for all users on planes_mantenimiento"
ON public.planes_mantenimiento FOR DELETE
USING (true);
